# BEGIN comment here
#
# RUN
# ack command line(s)
# They should NOT be shell-escaped.  Args are split on whitespace before
# being passed in to ack.
#
# YESLINES
# Lines that should match with underlines shown
#                              ^^^^^^^^^^
# YES
# Lines that should match, but without the underlines.
#
# NO
# Lines that should not match.
#
# END
#
# Blank lines are always ignored.



BEGIN Straight -w

RUN
ack -w foo

YESLINES
foo
^^^

End of the line foo
                ^^^

I pity da foo'.
          ^^^

NO
foobar
foot
underfoot

END


BEGIN optional character
RUN
ack foot?

YES
foo
foot
Trampled underfoot
foobarf
foo-bar
foo-bart
football

END


BEGIN -w and optional character
RUN
ack -w foot?

YES
foo
foot
foo-bar
foot-bar

YESLINES
foo
^^^

By the foot
       ^^^^

I pity da foo'.
          ^^^

NO
Trampled underfoot
football
foobar

END


BEGIN -w and optional group
RUN
ack -w foo(bar)?

NO

x foobar x
  ^^^^^^

                     ^^^^^^

END

END

          ^^^^^^^^^^^^^

NO
